44auto WatchdogDriver::init_impl(const std::uint32_t timeout) const
45 -> ErrorHandler<std::monostate, WatchdogError> {
46 if (!device_is_ready(wdt)) {
47 return ErrorHandler<std::monostate, WatchdogError>(
48 WatchdogError::DeviceNotReady, "Watchdog device not ready");
49 }

52 wdt_config.flags = WDT_FLAG_RESET_SOC;
53 wdt_config.window.min = 0;
54 wdt_config.window.max = timeout;
55
56 get_wdt_channel_id() = wdt_install_timeout(wdt, &wdt_config);
57
58 if (get_wdt_channel_id() == -ENOTSUP) {
59 wdt_config.callback = nullptr;
60 get_wdt_channel_id() = wdt_install_timeout(wdt, &wdt_config);
61 }
62
63 if (get_wdt_channel_id() < 0) {
64 return ErrorHandler<std::monostate, WatchdogError>(
65 WatchdogError::ConfigurationError,
66 "Failed to install watchdog timeout");
67 }
68
69 const auto setup_result{wdt_setup(wdt, WDT_OPT_PAUSE_HALTED_BY_DBG)};
70 if (setup_result != 0) {
71 return ErrorHandler<std::monostate, WatchdogError>(
72 errno_to_watchdog_error(setup_result), "Failed to setup watchdog");
73 }
74
75 return ErrorHandler<std::monostate, WatchdogError>(std::monostate{});
76}
