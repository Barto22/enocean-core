36auto Semaphore::do_init() -> ErrorHandler<std::monostate, SemaphoreError> {
37 if (is_initialized()) {
38 if (const int destroy_result{sem_destroy(&sem)}; destroy_result != 0) {
39 const int err{errno};
40 return ErrorHandler<std::monostate, SemaphoreError>(
41 errno_to_semaphore_error(err), "sem_destroy failed");
42 }
43 set_initialized(false);
44 }
45
46 if (const int result{sem_init(&sem, 0, get_initial_count())}; result == 0) {
47 set_initialized(true);
48 return ErrorHandler<std::monostate, SemaphoreError>(std::monostate{});
49 }
50 const int err{errno};
51 return ErrorHandler<std::monostate, SemaphoreError>(
52 errno_to_semaphore_error(err), "sem_init failed");
