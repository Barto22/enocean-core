63 // Args passed by value: all printf arguments are POD/primitive types
64 // and value semantics are correct for C variadic functions.
65 template <typename... Args>
66 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-vararg)
67 void operator()(const char* fmt, Args... args) const noexcept {
68 auto* const logger{plog::get<PLOG_DEFAULT_INSTANCE_ID>()};
69 if (logger == nullptr) {
70 return;
71 }
72 if (!logger->checkSeverity(severity)) {
73 return;
74 }
75 std::array<char, k_log_buf_size> buf{};
76 // When there are no format arguments the format string is passed
77 // through verbatim via "%s" to satisfy -Wformat-security, which only
78 // fires when a non-literal format string has no accompanying arguments.
79 if constexpr (sizeof...(args) == 0) {
80 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-vararg)
81 std::snprintf(buf.data(), buf.size(), "%s", fmt);
82 } else {
83 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-vararg, clang-diagnostic-format-nonliteral)
84 std::snprintf(buf.data(), buf.size(), fmt, args...);
85 }
86 *logger +=
87 plog::Record(severity, loc.function_name(),
88 static_cast<std::size_t>(loc.line()), loc.file_name(),
89 nullptr, PLOG_DEFAULT_INSTANCE_ID)
90 .ref()
91 << buf.data();
92 }
93};
