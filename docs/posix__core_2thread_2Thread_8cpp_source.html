38auto configure_priority(pthread_attr_t& attr, int priority) -> int {
39 if (priority < 0) {
40 (void)pthread_attr_setinheritsched(&attr, PTHREAD_INHERIT_SCHED);
41 return 0;
42 }
43
44 constexpr int policy{SCHED_FIFO};
45 const int min_priority{sched_get_priority_min(policy)};
46 const int max_priority{sched_get_priority_max(policy)};
47 if ((min_priority == -1) || (max_priority == -1)) {
48 (void)pthread_attr_setinheritsched(&attr, PTHREAD_INHERIT_SCHED);
49 return -EINVAL;
50 }
51
52 const int clamped{std::clamp(priority, min_priority, max_priority)};
53 sched_param sched{};
54 sched.sched_priority = clamped;
55
56 const bool policy_ok{pthread_attr_setschedpolicy(&attr, policy) == 0};
57 const bool param_ok{policy_ok &&
58 (pthread_attr_setschedparam(&attr, &sched) == 0)};
59 const bool explicit_ok{
60 param_ok &&
61 (pthread_attr_setinheritsched(&attr, PTHREAD_EXPLICIT_SCHED) == 0)};
62
63 if (policy_ok && param_ok && explicit_ok) {
64 return 0;
65 }
66
67 // Fallback to inherited scheduling when real-time priority is not allowed.
68 (void)pthread_attr_setschedpolicy(&attr, SCHED_OTHER);
69 (void)pthread_attr_setinheritsched(&attr, PTHREAD_INHERIT_SCHED);
70 return -EPERM;
71}

92auto sanitize_stack_size(std::size_t requested) -> std::size_t {
93 const auto min_stack{minimum_stack_size()};
94 if (requested == 0U) {
95 return min_stack;
96 }
97 const auto word_align{sizeof(void*) - 1U};
98 const auto aligned{(requested + word_align) & ~word_align};
99 return std::max(aligned, min_stack);
100}
