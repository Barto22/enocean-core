24auto Scanner::start_scan_impl() noexcept

56void Scanner::zephyr_scan_cb(const struct bt_le_scan_recv_info* info,
57 struct net_buf_simple* buf) noexcept {
58 Scanner* const inst{s_instance_.load(std::memory_order_acquire)};
59 if ((inst == nullptr) || (inst->callback_ == nullptr) ||
60 (info == nullptr) || (info->addr == nullptr) || (buf == nullptr)) {
61 return;
62 }
63
64 inst->callback_(
65 inst->callback_ctx_,
66 std::span<const std::uint8_t>(info->addr->a.val,
67 static_cast<std::size_t>(BT_ADDR_SIZE)),
68 info->addr->type, info->adv_type, info->rssi,
69 std::span<const std::uint8_t>(buf->data,
70 static_cast<std::size_t>(buf->len)));
