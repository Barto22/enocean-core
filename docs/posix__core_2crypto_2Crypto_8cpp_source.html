57bool AesCcm::aes_ccm_decrypt_impl(
58 std::span<const std::uint8_t> key, std::span<const std::uint8_t> nonce,
59 std::span<const std::uint8_t> aad, std::span<const std::uint8_t> ciphertext,
60 std::array<std::uint8_t, Enocean::k_tag_len> tag,
61 std::span<std::uint8_t> plaintext) noexcept {
62 EvpCtxGuard guard{};
63 if (!guard.valid()) {
64 LOGGER_ERROR("AesCcm: EVP_CIPHER_CTX_new failed");
65 return false;
66 }
67 EVP_CIPHER_CTX* ctx{guard.get()};
68
69 if (EVP_DecryptInit_ex(ctx, EVP_aes_128_ccm(), nullptr, nullptr, nullptr) !=
70 1) {
71 return false;
72 }
73
74 if (EVP_CIPHER_CTX_ctrl(ctx, EVP_CTRL_CCM_SET_IVLEN,
75 static_cast<int>(nonce.size()), nullptr) != 1) {
76 return false;
77 }
78
79 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-const-cast)
80 if (EVP_CIPHER_CTX_ctrl(ctx, EVP_CTRL_CCM_SET_TAG,
81 static_cast<int>(Enocean::k_tag_len),
82 static_cast<void*>(tag.data())) != 1) {
83 return false;
84 }
85
86 if (EVP_DecryptInit_ex(ctx, nullptr, nullptr, key.data(), nonce.data()) !=
87 1) {
88 return false;
89 }
90
91 int outl{0};
92
93 if (EVP_DecryptUpdate(ctx, nullptr, &outl, nullptr,
94 static_cast<int>(ciphertext.size())) != 1) {
95 return false;
96 }
97
98 if (!aad.empty()) {
99 if (EVP_DecryptUpdate(ctx, nullptr, &outl, aad.data(),
100 static_cast<int>(aad.size())) != 1) {
101 return false;
102 }
103 }
104
105 // For CCM, tag verification happens inside EVP_DecryptUpdate (not
106 // EVP_DecryptFinal_ex). A non-null output pointer is required even for
107 // auth-only (empty ciphertext) to trigger the internal CBC-MAC check.
108 // EVP_DecryptFinal_ex is intentionally not called.
109 std::array<std::uint8_t, 1U> dummy_out{};
110 std::uint8_t* const out_ptr{ciphertext.empty() ? dummy_out.data()
111 : plaintext.data()};
112
113 if (!ciphertext.empty() && (plaintext.size() < ciphertext.size())) {
114 LOGGER_ERROR("AesCcm: plaintext buffer too small");
115 return false;
116 }
117
118 const int ret{EVP_DecryptUpdate(
119 ctx, out_ptr, &outl,
120 ciphertext.empty() ? nullptr : ciphertext.data(),
121 static_cast<int>(ciphertext.size()))};
122 return ret >= 0;
123}
