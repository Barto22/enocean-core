30def modify_compilation_database(input_path: Path, output_path: Path):
31 """
32 Reads the compilation database, applies regex replacements,
33 and writes the modified database.
34 """
35 print("\nModifying compilation commands...")
36
37 if not input_path.exists():
38 print(f"Error: Input file not found at {input_path}")
39 sys.exit(1)
40
41 try:
42 with open(input_path, "r", encoding="utf-8") as f:
43 data = json.load(f)
44 except Exception as e:
45 print(f"Error reading {input_path}: {e}")
46 sys.exit(1)
47
48 replace_modules_ts = re.compile(r"-fmodules-ts")
49 remove_deps_format = re.compile(r"-fdeps-format=p1689r5")
50 remove_module_mapper = re.compile(r"-fmodule-mapper=[^\s]*")
51
52 modified_data = []
53
54 for entry in data:
55 command = entry.get("command")
56 if command:
57 command = replace_modules_ts.sub("-fmodules", command)
58 command = remove_deps_format.sub("", command)
59 command = remove_module_mapper.sub("", command)
60 command = re.sub(r"\s+", " ", command).strip()
61 entry["command"] = command
62
63 modified_data.append(entry)
64
65 try:
66 with open(output_path, "w", encoding="utf-8") as f:
67 json.dump(modified_data, f, indent=4)
68 print(f"Successfully created modified compilation database: {output_path}")
69 except Exception as e:
70 print(f"Error writing to {output_path}: {e}")
71 sys.exit(1)

85def is_codechecker_success(cmd: List[str], returncode: int) -> bool:
86 """
87 Determines whether a CodeChecker command finished successfully
88 based on its exit code and subcommand.
89 """
90 if len(cmd) < 2:
91 return returncode == 0
92
93 subcommand = cmd[1]
94
95 # CodeChecker analyze:
96 # 0 = success, no reports
97 # 3 = success, reports produced
98 if subcommand == "analyze":
99 return returncode in (0, 3)
100
101 # CodeChecker parse --export html may return non-zero even on success
102 if subcommand == "parse" and "--export" in cmd and "html" in cmd:
103 return returncode in (0, 1, 2)
104
105 return returncode == 0

108def run_subprocess(cmd: List[str]):
109 """Executes a subprocess command with CodeChecker-aware exit handling."""
110 cmd_str = " ".join(cmd)
111 print(f"\nExecuting command: {cmd_str}")
112
113 try:
114 result = subprocess.run(
115 cmd,
116 check=False,
117 capture_output=True,
118 text=True,
119 )
120
121 if is_codechecker_success(cmd, result.returncode):
122 print("Command executed successfully.")
123 return
124
125 print(f"Error: Command failed with exit code {result.returncode}")
126
127 if result.stdout:
128 print(f"STDOUT (last 500 chars):\n{result.stdout[-500:]}")
129 if result.stderr:
130 print(f"STDERR:\n{result.stderr}")
131
132 sys.exit(result.returncode)
133
134 except FileNotFoundError:
135 print("Error: CodeChecker command not found. Make sure CodeChecker is installed and in your PATH.")
136 sys.exit(1)
137 except Exception as e:
138 print(f"An unexpected error occurred during command execution: {e}")
139 sys.exit(1)

143 args = parse_arguments()

145 config_dir = Path(args.codechecker_config_dir).resolve()
146 build_dir = Path(args.build_dir).resolve()

148 input_json_path = build_dir / "compile_commands.json"

153 print("--- CodeChecker Analysis Automation ---")
154 print(f"Config Directory: {config_dir}")
155 print(f"Build Directory: {build_dir}")
156
157 # 1. Modify compilation database
158 modify_compilation_database(input_json_path, output_json_path)
159
160 # 2. Change directory to config dir
161 print(f"\nChanging working directory to: {config_dir}")
162 try:
163 os.chdir(config_dir)
164 except OSError as e:
165 print(f"Error changing directory: {e}")
166 sys.exit(1)
167
168 # 3. Clean output directories
169 print("\nCleaning up existing reports directories...")
170 clean_directory(reports_dir, "reports directory")
171 clean_directory(reports_html_dir, "HTML reports directory")
172
173 reports_dir.mkdir(parents=True, exist_ok=True)
174 reports_html_dir.mkdir(parents=True, exist_ok=True)
175
176 # 4. Run CodeChecker
177 analysis_file = output_json_path.as_posix()
178 reports_output = reports_dir.as_posix()
179 html_output = reports_html_dir.as_posix()
180
181 commands = [
182 ["CodeChecker", "analyze", analysis_file, "--config", "clang_tidy_config.json", "--output", reports_output],
183 ["CodeChecker", "analyze", analysis_file, "--config", "cppcheck_config.json", "--output", reports_output],
184 ["CodeChecker", "parse", "--export", "html", "--output", html_output, reports_output],
185 ]
186
187 for cmd in commands:
188 run_subprocess(cmd)
189
190 print("\n--- Analysis Complete ---")
191 print(f"Reports saved to: {reports_dir}")
192 print(f"HTML Report generated at: {reports_html_dir / 'index.html'}")
