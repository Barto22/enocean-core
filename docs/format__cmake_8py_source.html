21def matches_patterns(filename, patterns):
22 """
23 @brief Checks if a filename matches any of the given glob patterns.
24
25 @param filename The name of the file to check.
26 @param patterns A list of glob patterns (e.g., ["*.txt"]).
27 @return True if the filename matches any pattern, False otherwise.
28 """
29 return any(fnmatch.fnmatch(filename, pat) for pat in patterns)

31def find_cmake_files(root_dir, exclude_dirs):
32 """
33 @brief Recursively scans the root directory for CMake files, skipping excluded directories.
34
35 @param root_dir The starting directory for the search.
36 @param exclude_dirs A set of absolute paths to directories to skip.
37 @return A list of absolute paths to found CMake files.
38 """
39 cmake_files = []
40 for root, dirs, files in os.walk(root_dir):
41 # Exclude directories by modifying dirs in-place
42 dirs[:] = [d for d in dirs if os.path.join(root, d) not in exclude_dirs]
43
44 for f in files:
45 if matches_patterns(f, CMAKE_FILES_PATTERNS):
46 cmake_files.append(os.path.join(root, f))
47 return cmake_files

49def run_cmake_format(files, config=None, check_only=False, dry_run=False):
50 """
51 @brief Runs cmake-format on a list of files.
52
53 @param files A list of file paths to format.
54 @param config Optional path to a configuration file.
55 @param check_only If True, only check if files are formatted correctly without modifying.
56 @param dry_run If True, show what would be formatted without making changes.
57 @return A list of tuples containing (filename, exception/message) for any failures or issues.
58 """
59 errors = []
60 needs_formatting = []
61
62 for f in files:
63 if dry_run:
64 print(f"[would format] {f}")
65 continue
66
67 if check_only:
68 cmd = ["cmake-format", "--check", f]
69 else:
70 cmd = ["cmake-format", "-i", f]
71
72 if config:
73 cmd.extend(["--config", config])
74
75 try:
76 result = subprocess.run(cmd, check=False, capture_output=True, text=True)
77
78 if check_only:
79 if result.returncode != 0:
80 needs_formatting.append(f)
81 print(f"[needs formatting] {f}")
82 else:
83 print(f"[ok] {f}")
84 else:
85 if result.returncode == 0:
86 print(f"[formatted] {f}")
87 else:
88 errors.append((f, f"Formatting failed: {result.stderr}"))
89
90 except FileNotFoundError:
91 print("Error: 'cmake-format' executable not found in PATH.")
92 sys.exit(1)
93
94 if check_only and needs_formatting:
95 return needs_formatting
96
97 return errors
