98auto Scanner::start_scan_impl() noexcept
99 -> ErrorHandler<std::monostate, BleError> {
100 if (hci_fd_ < 0) {
101 return ErrorHandler<std::monostate, BleError>(
102 BleError::NotInitialised, "Call init() before start_scan()");

141int Scanner::open_hci_socket() noexcept {
142 const int fd{hci_open_dev(dev_id_)};
143 if (fd < 0) {
144 const int saved_errno{errno};
145 LOGGER_ERROR("hci_open_dev(%d) failed: %s", dev_id_,
146 errno_to_str(saved_errno));
147 return -1;
148 }
149
150 struct hci_filter flt{};
151 hci_filter_clear(&flt);
152 hci_filter_set_ptype(HCI_EVENT_PKT, &flt);
153 hci_filter_set_event(EVT_LE_META_EVENT, &flt);
154
155 if (::setsockopt(fd, SOL_HCI, HCI_FILTER, &flt, sizeof(flt)) < 0) {
156 const int saved_errno{errno};
157 LOGGER_ERROR("setsockopt(HCI_FILTER) failed: %s",
158 errno_to_str(saved_errno));
159 ::close(fd);
160 return -1;
161 }
162 return fd;
163}

165bool Scanner::enable_le_scan() noexcept {
166 // Passive scan, 100 ms interval, 100 ms window, own address = public,
167 // no duplicate filtering (show all so we don't miss any device).
168 constexpr std::uint16_t k_scan_interval{0x00A0U}; // 100 ms
169 constexpr std::uint16_t k_scan_window{0x00A0U}; // 100 ms
170 constexpr int k_scan_timeout_ms{1000};
171
172 // Disable scanning first — the controller rejects set_scan_parameters
173 // with EIO if a scan is already active (e.g. started by bluetoothd).
174 (void)hci_le_set_scan_enable(hci_fd_, 0x00U, 0x00U, k_scan_timeout_ms);
175
176 if (hci_le_set_scan_parameters(hci_fd_,
177 0x00U, // passive
178 k_scan_interval, k_scan_window,
179 0x00U, // own addr: public
180 0x00U, // no whitelist
181 k_scan_timeout_ms) < 0) {
182 const int saved_errno{errno};
183 LOGGER_ERROR("hci_le_set_scan_parameters failed: %s",
184 errno_to_str(saved_errno));
185 return false;
186 }
187
188 if (hci_le_set_scan_enable(hci_fd_, 0x01U, 0x00U, k_scan_timeout_ms) < 0) {
189 const int saved_errno{errno};
190 LOGGER_ERROR("hci_le_set_scan_enable failed: %s",
191 errno_to_str(saved_errno));
192 return false;
193 }
194 return true;
195}

211void Scanner::scan_loop() noexcept {
212 std::array<std::uint8_t, k_hci_buf_size> buf{};
213
214 while (running_.load(std::memory_order_acquire)) {
215 const ssize_t len{::read(hci_fd_, buf.data(), buf.size())};
216 if (len <= 0) {
217 if (!running_.load(std::memory_order_acquire)) {
218 break;
219 }
220 continue;
221 }
222
223 const auto total{static_cast<std::size_t>(len)};
224
225 // 5-byte HCI/LE-meta header + at least one full advertising report.
226 if (total < 5U + k_min_adv_report_size) {
227 continue;
228 }
229 if (buf[0U] != k_hci_event_pkt) {
230 continue;
231 }
232 if (buf[1U] != k_evt_le_meta) {
233 continue;
234 }
235 if (buf[3U] != k_le_adv_report_evt) {
236 continue;
237 }
238
239 const std::uint8_t num_reports{buf[4U]};
240 if (num_reports == 0U) {
241 continue;
242 }
243
244 dispatch_adv_report(buf.data() + 5U, total - 5U, num_reports);
245 }

248void Scanner::dispatch_adv_report(const std::uint8_t* report,
249 std::size_t report_buf_len,
250 std::uint8_t num_reports) noexcept {
251 if ((report == nullptr) || (callback_ == nullptr)) {
252 return;
253 }
254
255 // Sanity cap: the LE spec allows at most a handful of reports per event.
256 constexpr std::uint8_t k_max_reports{15U};
257 if (num_reports > k_max_reports) {
258 return;
259 }
260
261 const std::uint8_t* const buf_end{report + report_buf_len};
262 const std::uint8_t* ptr{report};
263
264 for (std::uint8_t r{0U}; r < num_reports; ++r) {
265 // Minimum record: type(1)+addr_type(1)+addr(6)+data_len(1) = 9 bytes,
266 // plus at least rssi(1) = 10 bytes total before reading data_len.
267 constexpr std::size_t k_report_hdr{10U};
268 if (ptr + k_report_hdr > buf_end) {
269 break;
270 }
271
272 const std::uint8_t adv_type{ptr[0U]};
273 const std::uint8_t addr_type{ptr[1U]};
274 const std::uint8_t data_len{ptr[8U]};
275
276 // Validate that data + rssi byte fit within the received buffer.
277 if (ptr + k_report_hdr + static_cast<std::size_t>(data_len) > buf_end) {
278 break;
279 }
280
281 const std::uint8_t* addr_ptr{ptr + 2U};
282 const std::uint8_t* data_ptr{ptr + 9U};
283 const auto* rssi_ptr{data_ptr + data_len};
284
285 const auto rssi{static_cast<std::int8_t>(*rssi_ptr)};
286
287 callback_(callback_ctx_,
288 std::span<const std::uint8_t>(addr_ptr,
289 static_cast<std::size_t>(6U)),
290 addr_type, adv_type, rssi,
291 std::span<const std::uint8_t>(
292 data_ptr, static_cast<std::size_t>(data_len)));
293
294 ptr = rssi_ptr + 1U;
295 }
