13auto MakeAbsoluteTimeout(std::chrono::milliseconds delta) -> timespec {
14 timespec ts{};
15 clock_gettime(CLOCK_REALTIME, &ts);
16
17 const auto delta_ns{
18 std::chrono::duration_cast<std::chrono::nanoseconds>(delta)};
19 const auto seconds{
20 std::chrono::duration_cast<std::chrono::seconds>(delta_ns)};
21 ts.tv_sec += static_cast<time_t>(seconds.count());
22
23 const auto extra_ns{static_cast<int64_t>(delta_ns.count() -
24 seconds.count() * 1000000000LL)};
25 ts.tv_nsec += extra_ns;
26 if (ts.tv_nsec >= 1000000000L) {
27 ts.tv_sec += 1;
28 ts.tv_nsec -= 1000000000L;
29 }
30 return ts;
31}

44TEST(PosixSemaphoreTests, InitialCountZeroBlocksUntilGive) {
45 PlatformCore::Semaphore sem("zero_init", 0, 1);
46 auto init_result{sem.init()};
47 ASSERT_TRUE(init_result.has_value());
48
49 std::atomic<bool> take_succeeded{false};
50 std::atomic<bool> thread_started{false};
51
52 std::thread waiter([&]() {
53 thread_started.store(true, std::memory_order_release);
54 auto result{sem.take(nullptr)};
55 take_succeeded.store(result.has_value(), std::memory_order_relaxed);
56 });
57
58 // Wait for thread to start and block
59 while (!thread_started.load(std::memory_order_acquire)) {
60 std::this_thread::yield();
61 }

63
64 // Semaphore should not have been acquired yet
65 EXPECT_FALSE(take_succeeded.load());
66
67 // Now give the semaphore
68 sem.give();
69
70 waiter.join();
71 EXPECT_TRUE(take_succeeded.load());

112TEST(PosixSemaphoreTests, MultipleThreadsWaitingForSemaphore) {
113 constexpr int num_threads{3};
114 PlatformCore::Semaphore sem("multi", 0, num_threads);
115 auto init_result{sem.init()};
116 ASSERT_TRUE(init_result.has_value());
117
118 std::atomic<int> success_count{0};
119 std::vector<std::thread> threads;
120 threads.reserve(num_threads);
121
122 for (int i = 0; i < num_threads; ++i) {
123 threads.emplace_back([&]() {
124 auto result{sem.take(nullptr)};
125 if (result.has_value()) {
126 success_count.fetch_add(1, std::memory_order_relaxed);
127 }
128 });
129 }
130
131 // Wait for all threads to start and block
132 std::this_thread::sleep_for(std::chrono::milliseconds{100});
133
134 // Release semaphore num_threads times
135 for (int i = 0; i < num_threads; ++i) {
136 sem.give();
137 std::this_thread::sleep_for(std::chrono::milliseconds{10});
138 }
139
140 for (auto& thread : threads) {
141 thread.join();
142 }
