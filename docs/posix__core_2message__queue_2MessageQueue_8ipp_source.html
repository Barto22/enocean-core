24template <typename T>
25auto MessageQueue<T>::init_impl(const std::uint32_t capacity)
26 -> ErrorHandler<std::monostate, MessageQueueError> {
27 struct mq_attr attr{};
28 attr.mq_flags = 0;
29 attr.mq_maxmsg = static_cast<long>(capacity);
30 attr.mq_msgsize = static_cast<long>(sizeof(T));
31 attr.mq_curmsgs = 0;
32
33 auto* queue_name{this->get_name()};
34 if ((queue_name == nullptr) || (queue_name[0] == '\0')) {
35 return ErrorHandler<std::monostate, MessageQueueError>(
36 MessageQueueError::InvalidParameter, "Queue name is empty");
37 }
38
39 // POSIX message queue names must start with '/'
40 if (queue_name[0] != '/') {
41 return ErrorHandler<std::monostate, MessageQueueError>(
42 MessageQueueError::InvalidParameter,
43 "Queue name must start with '/'");
44 }
45
46 mqd_ = mq_open(queue_name, O_CREAT | O_RDWR, 0644, &attr);
47
48 if (mqd_ == static_cast<mqd_t>(-1)) {
49 return ErrorHandler<std::monostate, MessageQueueError>(
50 errno_to_error(errno), "mq_open failed");
51 }
52 return ErrorHandler<std::monostate, MessageQueueError>(std::monostate{});
53}

55template <typename T>
56auto MessageQueue<T>::send_impl(T& msg, std::uint32_t timeout_ms) const
57 -> ErrorHandler<std::monostate, MessageQueueError> {
58 if (mqd_ == static_cast<mqd_t>(-1)) {
59 return ErrorHandler<std::monostate, MessageQueueError>(
60 MessageQueueError::InvalidQueue, "Queue not initialized");
61 }
62
63 constexpr std::uint32_t infinite_timeout{
64 std::numeric_limits<std::uint32_t>::max()};
65 int result{0};
66 if ((timeout_ms == 0U) || (timeout_ms == infinite_timeout)) {
67 result =
68 mq_send(mqd_, reinterpret_cast<const char*>(&msg), sizeof(T), 0U);
69 } else {
70 auto ts{ms_to_timespec(timeout_ms)};
71 result = mq_timedsend(mqd_, reinterpret_cast<const char*>(&msg),
72 sizeof(T), 0U, &ts);
73 }
74
75 if (result != 0) {
76 return ErrorHandler<std::monostate, MessageQueueError>(
77 errno_to_error(errno), "mq_send/mq_timedsend failed");
78 }
79 return ErrorHandler<std::monostate, MessageQueueError>(std::monostate{});

121template <typename T>
122auto MessageQueue<T>::ms_to_timespec(std::uint32_t timeout_ms) const
123 -> struct timespec {
124 struct timespec ts{};
125 clock_gettime(CLOCK_REALTIME, &ts);
126
127 const auto max_sec{std::numeric_limits<time_t>::max()};
128 const auto now_sec{ts.tv_sec};
129 const auto add_sec{static_cast<time_t>(timeout_ms / 1000U)};
130 const auto add_nsec{static_cast<long>((timeout_ms % 1000U) * 1000000U)};
131
132 if ((max_sec - now_sec) <= 0) {
133 ts.tv_sec = max_sec;
134 ts.tv_nsec = 999999999L;
135 return ts;
136 }
137
138 const auto safe_sec{(add_sec > (max_sec - now_sec)) ? (max_sec - now_sec)
139 : add_sec};
140 ts.tv_sec += safe_sec;
141 ts.tv_nsec += add_nsec;
142 if (ts.tv_nsec >= 1000000000L) {
143 if (ts.tv_sec < max_sec) {
144 ts.tv_sec++;
145 ts.tv_nsec -= 1000000000L;
146 } else {
147 ts.tv_nsec = 999999999L;
148 }
149 }
150
151 return ts;
152}
