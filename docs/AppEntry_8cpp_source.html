97 [[maybe_unused]] PlatformCore::ThreadParamVariant& params) {
98 LOGGER_NOTICE("Application starting...");
99
100 Crypto::AesCcm crypto{};
101
102 const Enocean::EnoceanCallbacks callbacks{
103 .button = on_button,
104 .sensor = on_sensor,
105 .commissioned = on_commissioned,
106 .decommissioned = on_decommissioned,
107 .loaded = on_loaded,
108 };
109
110 Enocean::EnoceanDriver<Crypto::AesCcm> driver{crypto, callbacks};
111
112 const auto init_result{driver.init()};
113 if (!init_result) {
114 LOGGER_ERROR("EnOcean driver init failed");
115 return;
116 }
117
118 Ble::Scanner scanner{};
119
120 const auto ble_init{scanner.init(
121 &Enocean::EnoceanDriver<Crypto::AesCcm>::advertisement_cb, &driver)};
122 if (!ble_init) {
123 LOGGER_ERROR("BLE scanner init failed");
124 return;
125 }
126
127 const auto scan_start{scanner.start_scan()};
128 if (!scan_start) {
129 LOGGER_ERROR("BLE scan start failed");
130 return;
131 }
132
133 driver.enable_commissioning();
134
135 LOGGER_NOTICE("EnOcean scanner ready - commissioning enabled.");
136
137 PlatformCore::Sleep sleep{};
